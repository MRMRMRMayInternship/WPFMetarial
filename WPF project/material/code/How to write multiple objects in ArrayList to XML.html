      Alternatively you can use DataContrract Serializer of System.Runtime.Serialization namespace.

      For example you have calss Person and List of people class

       [DataContract(Name = "People", Namespace = "Your Project Namespace")]
 public class People
 {
  [DataMember]
  public IList<Person> people = new List<Person>();
 }

 public class Person
 {
  public string Name { get; set; }
  public int Age { get; set; }
 }

      The can serialize and deserialize

         //Creating the list
   People pl = new People();
   pl.people.Add(new Person{Name="Person1",Age=20});
   pl.people.Add(new Person{Name="Person2",Age=30});

   //Serialization
   FileStream fs1 = new FileStream(Application.StartupPath + @"\test2.xml", FileMode.OpenOrCreate);
   DataContractSerializer ser = new DataContractSerializer(pl.GetType());
   ser.WriteObject(fs1, pl);
   fs1.Close();

   //deserialization
   FileStream fs2 = new FileStream(Application.StartupPath + @"\test2.xml", FileMode.Open);
   People pl2 = (People)ser.ReadObject(fs2);
   fs2.Close();

      The error("cannot create XML file") may be probably because  the streamwriter is inside the loop and you did not close it. 

      I'm  not very bright with this , but this worked for me  :

       

       

       

      StreamWriter writer = new StreamWriter("test1.xml");

            foreach (object o in obj.ToArray())
            {
                System.Xml.Serialization.XmlSerializer x = new System.Xml.Serialization.XmlSerializer(o.GetType());
                x.Serialize(writer, o);
            }
            writer.Close();   

       

       

       

       

      Hope it help :)

      Below code works for me

      using System.Collections;
using System.IO;
using System.Xml;
using System.Xml.Serialization;

namespace ConsoleApplication1
{
  class Program
  {

    static void Main(string[] args)
    {
      AListWrapper wrapper = new AListWrapper();

      XmlSerializer mySerializer = new XmlSerializer(typeof(AListWrapper));

      StreamWriter myWriter = new StreamWriter("c:\\myFileName.xml");

      mySerializer.Serialize(myWriter, wrapper);
      myWriter.Close();
    }
  }

  public class Animal
  {
    public string Type { get; set; }
    public int Age { get; set; }
  }

  public class Employee
  {
    public string Name { get; set; }
    public double Salary { get; set; }
    public string Address { get; set; }
  }

  [XmlRoot("ArrayList")]
  public class AListWrapper
  {
    [XmlElement(Type = typeof(Employee)),
    XmlElement(Type = typeof(Animal))]
    public ArrayList list = new ArrayList();

    public AListWrapper()
    {
      Animal animal = new Animal()
      {
        Age = 1,
        Type = "Dog"
      };

      Employee emp = new Employee()
      {
        Address = "Address",
        Name = "SomeName",
        Salary = 2000.50
      };

      list.Add(animal);
      list.Add(emp);
    }
  }
}

      Generated XML

      <?xml version="1.0" encoding="utf-8"?>
<ArrayList xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xmlns:xsd="http://www.w3.org/2001/XMLSchema">
 <Animal>
  <Type>Dog</Type>
  <Age>1</Age>
 </Animal>
 <Employee>
  <Name>SomeName</Name>
  <Salary>2000.5</Salary>
  <Address>Address</Address>
 </Employee>
</ArrayList>
